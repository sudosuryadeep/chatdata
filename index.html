Pehli baar setup: Apna Supabase project connect karo. Isse saara data permanently save hoga — refresh pe logout nahi hoga.

Pehle Supabase mein ye SQL run karo:

-- 1. Profiles table create table profiles ( id uuid references auth.users primary key, name text, role text default 'admin', avatar text, created_at timestamptz default now() ); alter table profiles enable row level security; create policy "own" on profiles for all using (auth.uid()=id); create policy "sa_read" on profiles for select using (true); -- 2. Bots table create table bots ( id uuid primary key default gen_random_uuid(), owner_id uuid references profiles(id) on delete cascade, token text unique not null, bot_username text, bot_name text, category text, description text, is_active bool default true, last_update_id bigint default 0, created_at timestamptz default now() ); alter table bots enable row level security; create policy "own" on bots for all using ( owner_id=auth.uid() or exists(select 1 from profiles where id=auth.uid() and role='superadmin') ); -- 3. Telegram users create table tg_users ( id uuid primary key default gen_random_uuid(), bot_id uuid references bots(id) on delete cascade, chat_id text not null, name text, username text, lang_code text, is_blocked bool default false, msg_count int default 0, created_at timestamptz default now(), unique(bot_id, chat_id) ); alter table tg_users enable row level security; create policy "via_bot" on tg_users for all using ( exists(select 1 from bots where id=bot_id and ( owner_id=auth.uid() or exists(select 1 from profiles where id=auth.uid() and role='superadmin') )) ); -- 4. Messages create table messages ( id uuid primary key default gen_random_uuid(), bot_id uuid references bots(id) on delete cascade, tg_user_id uuid references tg_users(id) on delete cascade, chat_id text, text text, media_type text, is_outgoing bool default false, tg_msg_id bigint, created_at timestamptz default now() ); alter table messages enable row level security; create policy "via_bot" on messages for all using ( exists(select 1 from bots where id=bot_id and ( owner_id=auth.uid() or exists(select 1 from profiles where id=auth.uid() and role='superadmin') )) ); -- 5. Trigger: auto-create profile on signup create or replace function handle_new_user() returns trigger as $$ begin insert into profiles(id, name) values(new.id, coalesce(new.raw_user_meta_data->>'name', split_part(new.email,'@',1))); return new; end; $$ language plpgsql security definer; create trigger on_auth_user_created after insert on auth.users for each row execute procedure handle_new_user();
