📋 Supabase SQL — First Time Setup

-- Run this in Supabase → SQL Editor:

create table if not exists admins (
  id bigint primary key generated always as identity,
  auth_id uuid references auth.users(id) on delete cascade,
  name text, email text, bot_token text,
  purpose text, theme text, bot_info text,
  is_banned boolean default false,
  created_at timestamptz default now()
);
create table if not exists bot_users (
  id bigint primary key generated always as identity,
  admin_id bigint references admins(id) on delete cascade,
  chat_id text, name text, username text,
  lang_code text, chat_type text,
  unread_count int default 0,
  is_blocked boolean default false,
  updated_at timestamptz default now(),
  unique(admin_id, chat_id)
);
create table if not exists messages (
  id bigint primary key generated always as identity,
  admin_id bigint references admins(id) on delete cascade,
  user_id bigint references bot_users(id) on delete cascade,
  tg_message_id bigint, text text, msg_type text,
  date_ts bigint, is_out boolean default false,
  unique(admin_id, tg_message_id)
);
create table if not exists broadcast_log (
  id bigint primary key generated always as identity,
  admin_id bigint references admins(id) on delete cascade,
  message text, sent_count int, fail_count int,
  created_at timestamptz default now()
);
-- Enable RLS
alter table admins enable row level security;
alter table bot_users enable row level security;
alter table messages enable row level security;
alter table broadcast_log enable row level security;
-- Policies (admins see only their own data)
create policy "own_admins" on admins for all using (auth.uid()=auth_id);
create policy "own_users" on bot_users for all using (admin_id in (select id from admins where auth_id=auth.uid()));
create policy "own_msgs" on messages for all using (admin_id in (select id from admins where auth_id=auth.uid()));
create policy "own_bc" on broadcast_log for all using (admin_id in (select id from admins where auth_id=auth.uid()));
